-- Functionality:

local sections = {
    LocalPlayerButton = LocalPlayerContent,
    GameButton = GameContent,
    ScriptsButton = ScriptsContent,
    ExploitButton = ExploitContent,
    InfoButton = InfoContent
}

local function showSection(sectionName)
    for _, section in pairs(sections) do
        section.Visible = false
    end
    sections[sectionName].Visible = true
end

for buttonName, _ in pairs(sections) do
    local button = Sidebar:FindFirstChild(buttonName)
    if button then
        button.MouseButton1Click:Connect(function()
            showSection(buttonName)
        end)
    end
end

Exit.MouseButton1Click:Connect(function()
    OrionScriptHub:Destroy()
end)

Clear.MouseButton1Click:Connect(function()
    ScriptHolder.Text = ""
end)

Execute.MouseButton1Click:Connect(function()
    local script = ScriptHolder.Text
    loadstring(script)()
end)
